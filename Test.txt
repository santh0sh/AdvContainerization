import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.data.jpa.repository.JpaRepository;

@Repository
public interface PressReleaseRepository extends JpaRepository<PressReleaseEntity, String> {
    
    @Query("SELECT pr FROM PressReleaseEntity pr ORDER BY pr.publishedDate DESC")
    Page<PressReleaseEntity> findPressReleases(Pageable pageable);
}
