CREATE TABLE informatical_articles (
    prid            VARCHAR2(32) PRIMARY KEY DEFAULT UPPER(RAWTOHEX(SYS_GUID())), -- Unique ID (GUID)
    title           VARCHAR2(255) NOT NULL, -- Main title of the article
    subtitle        VARCHAR2(255), -- Subtitle (optional)
    category        VARCHAR2(100) NOT NULL, -- Category of the article
    content         CLOB NOT NULL, -- Full content of the article
    thumbnail       BLOB, -- Image thumbnail (optional)
    publish_time    TIMESTAMP DEFAULT SYSTIMESTAMP, -- Publishing time (default: current time)
    send_email      CHAR(1) CHECK (send_email IN ('Y', 'N')) DEFAULT 'N', -- Email notification flag
    created_at      TIMESTAMP DEFAULT SYSTIMESTAMP, -- Record creation timestamp
    updated_at      TIMESTAMP DEFAULT SYSTIMESTAMP ON UPDATE CURRENT_TIMESTAMP -- Last updated timestamp
);
