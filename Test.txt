public interface TagRepository extends JpaRepository<TagEntity, String> {

    List<TagEntity> findByTagTypeOrderByTagNameAsc(String tagType);
}

public List<TagDTO> getTagsByType(String tagType) {
    return tagRepository.findByTagTypeOrderByTagNameAsc(tagType)
        .stream()
        .map(tag -> new TagDTO(tag.getTagId(), tag.getTagName(), tag.getTagDesc(), tag.getTagType()))
        .collect(Collectors.toList());
}
