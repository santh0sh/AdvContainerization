SELECT 
    PAGE_ID,
    COMP_LABEL,
    VALUE_GRP_COUNTER,
    LISTAGG(ATTR_NAME || '=' || ATTR_VALUE, ', ') 
        WITHIN GROUP (ORDER BY ATTR_NAME) AS ATTRIBUTES
FROM TB_MR_COMP_ATTR_DATA_VALUES
WHERE PAGE_ID = :your_page_id
GROUP BY 
    PAGE_ID, 
    COMP_LABEL, 
    VALUE_GRP_COUNTER
ORDER BY 
    COMP_LABEL, 
    VALUE_GRP_COUNTER;
